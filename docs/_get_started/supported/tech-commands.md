```bash
# Force scanning to techs 
noir -t rails

# Show all techs
noir --list-techs

#  TECHNOLOGIES:
#    -t TECHS, --techs rails,php      Specify the technologies to use
#    --exclude-techs rails,php        Specify the technologies to be excluded
#    --list-techs                     Show all technologies
```
